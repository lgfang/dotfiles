*** Swap spaces between displays

#+begin_src bash :tangle ~/.config/yabai/swap-spaces
  #!/usr/bin/env bash
  # Created: Fang Lungang 2024-05-17
  # Updated: Fang Lungang <<time-stamp-of-update()>>

  # Move currently visible spaces between displays. Useful when I move from home
  # to office, or vice versa.

  set -o errexit
  set -o pipefail
  PATH=$HOME/.local/bin:/opt/homebrew/bin:$PATH

  # Store current window ID to restore focus later
  current_window=$(yabai -m query --windows --window | jq '.id')

  # the array of visible spaces
  for i in $(yabai -m query --spaces | jq '.[]|select(."is-visible").index'); do
      visible_spaces+=($i)
  done

  # Rotate the array since I don't know the relative positions of the spaces
  # (displays). The first space will swap with every other space until the end of
  # this array. That said, since most of the time I have only two displays, this
  # script should work for me.
  array_length=${#visible_spaces[@]}
  for ((i=1; i<array_length; i++)); do
      yabai -m space "${visible_spaces[$((i-1))]}" --swap "${visible_spaces[$((i))]}"
  done

  # Restore focus to the original window
  if [ -n "$current_window" ] && [ "$current_window" != "null" ]; then
      yabai -m window --focus "$current_window"
  fi
#+end_src

*** Swap window across displays

#+begin_src bash :tangle  ~/.config/yabai/swap-windows
  #!/usr/bin/env bash
  # Created: Fang Lungang 2024-01-25
  # Updated: Fang Lungang <<time-stamp-of-update()>>

  set -o errexit
  set -o pipefail
  PATH=$HOME/.local/bin:/opt/homebrew/bin:$PATH
  target="$1"
  if [ "$target" == "west" ]; then
      opposite="east"
      window_placement="second_child"
  elif [ "$target" == "east" ]; then
      opposite="west"
      window_placement="first_child"
  else
      echo "Unspported direction '$target'" >&2
      exit 1
  fi

  current_window=$(yabai -m query --windows --window | jq '.id')

  if [ "$(yabai -m query --windows --display "$target" | jq 'length')" == "0" ]; then
      # The other display has no window to swap, just move the current window to it.
      yabai -m window --display "$target"
  else
      # the other display has window(s), swap the one next to the display border

      # move to the next display
      yabai -m display --focus "$target"

      # find the window on the border
      while yabai -m window --focus "$opposite" 2>/dev/null; do
          true
      done

      to_swap=$(yabai -m query --windows --window | jq '.id')
      original_window_placement=$(yabai -m config window_placement)

      yabai -m config window_placement "$window_placement"
      yabai -m window "$to_swap" --swap "$current_window"
      yabai -m config window_placement "$original_window_placement"
  fi

  # Ensure focus returns to the original window
  if [ -n "$current_window" ] && [ "$current_window" != "null" ]; then
    yabai -m window --focus "$current_window"
  fi
#+end_src

  # Created: Fang Lungang 2024-01-22
  # Updated: Fang Lungang <<time-stamp-of-update()>>

  set -o errexit
  set -o pipefail

  # a safeguard, ensure no inifite loop
  i=8

  win=$(yabai -m query --windows --window first | jq '.id')
  while yabai -m window $win --swap next &> /dev/null && [ $i -gt 0 ]; do
      i=$((i-1))
  done

* AeroSpace

I prefer AeroSpace over Yabai for the following reasons:
- Superior stability, especially when connecting or disconnecting external
  monitors while moving between workplaces.
- Proper window management for GUI Emacs.
- Native application launching, which eliminates the need to configure Karabiner
  hotkeys to open or focus apps like Slack and Chrome.

** Script to swap windows cross monitor boundaries

#+begin_src bash :tangle ~/.config/aerospace/aerospace-swap-windows.sh :shebang "#!/usr/bin/env bash"
  # Ensure aerospace can be found when the script is run without custom PATH
  export PATH="/opt/homebrew/bin:/usr/local/bin:$PATH"

  DIRECTION=$1

  # Get the ID and workspace of the currently focused window
  ID1=$(aerospace list-windows --focused --format "%{window-id}")
  WS1=$(aerospace list-windows --focused --format "%{workspace}")

  # Focus the window in the target direction (allowing monitor crossing)
  aerospace focus --boundaries all-monitors-outer-frame $DIRECTION

  # Get the ID and workspace of the newly focused window
  ID2=$(aerospace list-windows --focused --format "%{window-id}")
  WS2=$(aerospace list-windows --focused --format "%{workspace}")

  # If the window didn't change (e.g., edge of the screen), do nothing
  if [ "$ID1" == "$ID2" ]; then
      exit 0
  fi

  if [ "$WS1" == "$WS2" ]; then
      # If they are in the same workspace, use the native move command
      aerospace focus --window-id $ID1
      aerospace move $DIRECTION
  else
      # Explicitly focus and move the first window
      aerospace focus --window-id $ID1
      aerospace move-node-to-workspace $WS2

      # Explicitly focus and move the second window
      aerospace focus --window-id $ID2
      aerospace move-node-to-workspace $WS1

      # Restore focus to the original window
      aerospace focus --window-id $ID1
  fi
#+end_src

** Script to swap workspaces

#+begin_src bash :tangle ~/.config/aerospace/aerospace-swap-workspaces.sh :shebang "#!/usr/bin/env bash"
  # Swap the focused workspace with the visible workspace on the neighboring
  # monitor in the given direction.
  # Usage: swap-workspace-with-monitor.sh <left|right|up|down|next|prev>

  # Ensure aerospace can be found when the script is run without custom PATH
  export PATH="/opt/homebrew/bin:/usr/local/bin:$PATH"

  set -euo pipefail

  direction="${1:-}"
  case "$direction" in
    left|right|up|down|next|prev) ;;
    ,*)
      echo "Usage: $(basename "$0") <left|right|up|down|next|prev>" >&2
      exit 2
      ;;
  esac

  src_mon=$(aerospace list-monitors --focused --format '%{monitor-id}')
  src_ws=$(aerospace list-workspaces --focused)

  # Probe the neighbor by briefly moving focus there.
  aerospace focus-monitor "$direction"
  dst_mon=$(aerospace list-monitors --focused --format '%{monitor-id}')

  if [[ "$dst_mon" == "$src_mon" ]]; then
    # No monitor in that direction — restore focus and bail.
    aerospace workspace "$src_ws"
    echo "No monitor '$direction' of monitor $src_mon" >&2
    exit 0
  fi

  dst_ws=$(aerospace list-workspaces --focused)

  # Swap.
  aerospace workspace "$src_ws"
  aerospace move-workspace-to-monitor "$dst_mon"
  aerospace workspace "$dst_ws"
  aerospace move-workspace-to-monitor "$src_mon"

  # Land on the originally-focused workspace (now on the neighbor monitor).
  aerospace workspace "$src_ws"
#+end_src

** Script to jump to / summon an application

#+begin_src bash :tangle ~/.config/aerospace/aerospace-app.sh :shebang "#!/usr/bin/env bash"
  # Jump to or summon an application:
  #   - jump:   focus its (first) window wherever it lives, switching to
  #             that window's workspace as needed.
  #   - summon: move all of its windows to the focused workspace and focus
  #             the first one.
  #   - If no window matches, run the fallback command when given,
  #     otherwise launch the app with `open -a`.
  # Usage: aerospace-app.sh <jump|summon> <app-name> [prefix [fallback-cmd...]]
  # Example:
  #   aerospace-app.sh jump Slack
  #   aerospace-app.sh summon WezTerm '!ip-10-'   # local WezTerm only
  #   aerospace-app.sh jump WezTerm ip-10- \
  #       wezterm cli spawn --new-window -- ssh ws
  #
  # When [prefix] is given, only windows whose title starts with the prefix
  # (case-insensitive) are considered. Prefix the argument with "!" to
  # negate the match. This lets you distinguish multiple windows of the same
  # app — e.g. local vs remote WezTerm, whose tmux titles contain the
  # hostname (see set-titles-string in tmux.conf). A freshly launched
  # WezTerm window has no tmux title yet, so it matches the negated
  # (local) filter — which is the desired behavior.
  #
  # The fallback command matters for positive-prefix filters: for "remote
  # WezTerm" (prefix ip-10-), `open -a WezTerm` would open a *local* window
  # that still wouldn't match, so pass a command that creates the remote
  # session instead (see the Example above).

  # Why this exists (the bug it fixes):
  #   The jump bindings used to call `open -a <app>` directly. macOS treats
  #   `open -a` as a no-op when the target app is ALREADY the frontmost
  #   application -- it won't raise a window or make aerospace switch to the
  #   app's workspace. That produced a confusing "can't jump back" symptom:
  #     1. Focus is on a WezTerm window in workspace 0, so macOS considers
  #        WezTerm the frontmost app.
  #     2. Summon an EMPTY workspace (e.g. 9). Aerospace switches to it, but
  #        because that workspace has no window to take focus, WezTerm REMAINS
  #        the macOS-frontmost app (its window is just off-screen on ws 0).
  #     3. Jump to WezTerm -> `open -a WezTerm` sees it already frontmost and
  #        does nothing. Nothing happens; you appear stuck on the empty ws.
  #     4. Jump to Chrome -> Chrome is NOT frontmost, so `open -a` activates it
  #        and aerospace follows focus to Chrome's workspace. Works.
  #     5. Jump to WezTerm again -> WezTerm is no longer frontmost, so now
  #        `open -a` works.
  #   Note the empty-workspace detail: if the summoned workspace already has
  #   windows, focusing one of them changes the frontmost app away from
  #   WezTerm, so `open -a WezTerm` works and the bug never appears.
  #
  #   Using `aerospace focus --window-id` instead sidesteps macOS frontmost
  #   state entirely: aerospace focuses the window (and switches to its
  #   workspace) based on its own window tracking, so jumping back always works.

  export PATH="/opt/homebrew/bin:/usr/local/bin:$PATH"
  set -euo pipefail

  MODE="${1:-}"
  APP_NAME="${2:-}"
  if [ -z "$APP_NAME" ] || { [ "$MODE" != jump ] && [ "$MODE" != summon ]; }; then
    echo "Usage: $(basename "$0") <jump|summon> <app-name> [prefix [fallback-cmd...]]" >&2
    exit 2
  fi
  shift 2
  PREFIX="${1:-}"
  if [ $# -gt 0 ]; then shift; fi
  # Remaining "$@" is the optional fallback command.

  # Determine whether the prefix filter is negated (starts with "!").
  NEGATE=0
  PFX="$PREFIX"
  if [[ "$PFX" == !* ]]; then
    NEGATE=1
    PFX="${PFX#!}"
  fi

  # Window IDs whose app-name (and optional title prefix) matches.
  # The "|" delimiter avoids issues with app names containing spaces.
  WINDOW_IDS=$(aerospace list-windows --all --format "%{window-id}|%{app-name}|%{window-title}" \
                 | awk -F'|' -v app="$APP_NAME" -v pfx="$PFX" -v neg="$NEGATE" '
          tolower($2) == tolower(app) {
              if (pfx == "") { print $1; next }
              starts = (index(tolower($3), tolower(pfx)) == 1)
              if ((neg == 0 && starts) || (neg == 1 && !starts)) print $1
          }')

  if [ -z "$WINDOW_IDS" ]; then
    if [ $# -gt 0 ]; then
      exec "$@"
    fi
    open -a "$APP_NAME"
    exit 0
  fi

  FIRST_WID=$(head -n1 <<< "$WINDOW_IDS")

  if [ "$MODE" = summon ]; then
    CURRENT_WS=$(aerospace list-workspaces --focused)
    while IFS= read -r wid; do
      [ -z "$wid" ] && continue
      aerospace move-node-to-workspace --window-id "$wid" "$CURRENT_WS"
    done <<< "$WINDOW_IDS"
  fi

  aerospace focus --window-id "$FIRST_WID"
#+end_src
